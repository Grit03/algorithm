function solution(files) {
    const re = /^([a-zA-Z-\. ]+)([0-9]+)(.*)$/
    let dict = []
    files.forEach((entry, idx) => {
        let [fn, head, num] = entry.match(re)
        dict.push({fn, head: head.toLowerCase(), num: parseInt(num), idx})
    })

    return dict.sort((a, b) => {
        if (a.head > b.head) return  1
        if (a.head < b.head) return -1
        if (a.num > b.num) return  1
        if (a.num < b.num) return -1
        return a.idx - b.idx
    }).map(e => e.fn)
}

    // HEAD 부분 사전 순(대소문자 구분 없음)
    // HEAD가 같으면 숫자 순으로 정렬 (숫자 앞 0은 무시)
    // 나머지는 입력 순서대로 순서 유지
    
    const answer = files.sort((a, b) => {
        const [_a, a_head, a_number, a_tail] = a.toLowerCase().match(/([\D]+)(\d{1,5})(.+)/);
        const [_b, b_head, b_number, b_tail] = b.toLowerCase().match(/([\D]+)(\d{1,5})(.+)/);
        
        // abc
        // abcde
        if(a_head !== b_head){
            const longest = Math.max(a_head.length, b_head.length);
            for(let i =0; i <longest; i++){
                if(a_head[i] === undefined) return -1;
                if(b_head[i] === undefined) return 1;
                if(a_head.charAt(i) !== b_head.charAt(i))
                    return a_head.charAt(i) < b_head.charAt(i) ? -1 : 1;
            }
        }
        if(Number(a_number) !== Number(b_number)){
            return Number(a_number) - Number(b_number);
        }
        return 0;
    });
    
    return answer;
}
